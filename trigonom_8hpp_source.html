170 template <typename T> constexpr T archimedes(const T& q_1, const T& q_2, const T& q_3) {
171 validate_no_overflow_multiply(q_1, q_2, "q_1 in archimedes");
172 validate_no_overflow_multiply(q_1, T(4), "4 * q_1 in archimedes");
173 validate_no_overflow_multiply(q_2, T(4), "4 * q_2 in archimedes");

503 template <typename T> constexpr Triangle<T> create_triangle(const std::vector<T>& a,
504 const std::vector<T>& b,
505 const std::vector<T>& c) {
506 validate_vector_size(a, 2);
507 validate_vector_size(b, 2);
508 validate_vector_size(c, 2);
509
510 Triangle<T> tri;
511 tri.a = a;
512 tri.b = b;
513 tri.c = c;
514
515 // Calculate side vectors
516 std::vector<T> ab = {b[0] - a[0], b[1] - a[1]};
517 std::vector<T> bc = {c[0] - b[0], c[1] - b[1]};
518 std::vector<T> ca = {a[0] - c[0], a[1] - c[1]};
519
520 // Calculate quadrances
521 tri.q_ab = quad(ab);
522 tri.q_bc = quad(bc);
523 tri.q_ca = quad(ca);
524
525 // Calculate quadrea
526 tri.quadrea = archimedes(tri.q_ab, tri.q_bc, tri.q_ca);
527
528 // Validate triangle is non-degenerate
529 validate_non_zero(tri.quadrea, "quadrea (collinear points)");
530
531 // Calculate spreads
532 tri.spread_a = spread_law(tri.q_ca, tri.q_ab, tri.q_bc);
533 tri.spread_b = spread_law(tri.q_ab, tri.q_bc, tri.q_ca);
534 tri.spread_c = spread_law(tri.q_bc, tri.q_ca, tri.q_ab);
535
536 return tri;
537 }

549 template <typename T>
550 constexpr Triangle<T> solve_triangle_sas(const T& q_1, const T& q_2, const T& spread) {
551 validate_non_negative(q_1, "q_1");
552 validate_non_negative(q_2, "q_2");
553 validate_non_negative(spread, "spread");
554 validate_non_zero(q_1, "q_1");
555 validate_non_zero(q_2, "q_2");
556 validate_non_negative(spread, "spread");
557
558 // Calculate third quadrance using triple quad formula
559 // Note: triple_quad_formula returns the squared third quadrance
560 T q_3_squared = triple_quad_formula(q_1, q_2, spread);
561
562 // Take square root to get the actual third quadrance
563 if constexpr (std::is_floating_point_v<T>) {
564 if (q_3_squared < 0) {
565 throw TrigonomError("Invalid triangle: third quadrance squared is negative");
566 }
567 }
568
569 T q_3 = std::sqrt(q_3_squared);
570
571 // Verify triangle is valid (positive quadrea)
572 T q = archimedes(q_1, q_2, q_3);
573 if (q < T(0)) {
574 throw TrigonomError("Invalid triangle: quadrea is negative for SAS case");
575 }
576
577 // Calculate other spreads
578 T spread_2 = spread_law(q_2, q_3, q_1);
579 T spread_3 = spread_law(q_3, q_1, q_2);
580
581 // Create triangle with vertex at origin
582 Triangle<T> tri;
583 tri.a = {T(0), T(0)};
584 tri.b = {q_1, T(0)}; // Side AB along x-axis
585 tri.c = {q_1 * (1 - spread), std::sqrt(q_1 * q_2 * spread)}; // Point C
586
587 tri.q_ab = q_1;
588 tri.q_bc = q_2;
589 tri.q_ca = q_3;
590 tri.quadrea = q;
591 tri.spread_a = spread_3;
592 tri.spread_b = spread_2;
593 tri.spread_c = spread;
594
595 return tri;
596 }
